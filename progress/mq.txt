using IBM.WMQ;
using System.Collections;

public class MqProducer
{
    public void Send()
    {
        var props = new Hashtable
        {
            { MQC.HOST_NAME_PROPERTY, "localhost" },
            { MQC.PORT_PROPERTY, 1414 },
            { MQC.CHANNEL_PROPERTY, "DEV.APP.SVRCONN" },
            { MQC.USER_ID_PROPERTY, "app" },
            { MQC.PASSWORD_PROPERTY, "passw0rd" },
            { MQC.TRANSPORT_PROPERTY, MQC.TRANSPORT_MQSERIES_CLIENT }
        };

        using var qMgr = new MQQueueManager("QM1", props);

        using var queue = qMgr.AccessQueue(
            "DEV.QUEUE.1",
            MQC.MQOO_OUTPUT | MQC.MQOO_FAIL_IF_QUIESCING);

        var message = new MQMessage();
        message.WriteString("Hello from IBMMQDotnetClient!");

        var putOptions = new MQPutMessageOptions();
        queue.Put(message, putOptions);
    }
}



using IBM.WMQ;
using System.Collections;

public class Worker : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        var props = new Hashtable
        {
            { MQC.HOST_NAME_PROPERTY, Environment.GetEnvironmentVariable("MQ_HOST") ?? "localhost" },
            { MQC.PORT_PROPERTY, int.Parse(Environment.GetEnvironmentVariable("MQ_PORT") ?? "1414") },
            { MQC.CHANNEL_PROPERTY, "DEV.APP.SVRCONN" },
            { MQC.USER_ID_PROPERTY, "app" },
            { MQC.PASSWORD_PROPERTY, "passw0rd" },
            { MQC.TRANSPORT_PROPERTY, MQC.TRANSPORT_MQSERIES_CLIENT }
        };

        using var qMgr = new MQQueueManager("QM1", props);

        using var queue = qMgr.AccessQueue(
            "DEV.QUEUE.1",
            MQC.MQOO_INPUT_AS_Q_DEF | MQC.MQOO_FAIL_IF_QUIESCING);

        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                var msg = new MQMessage();

                var getOptions = new MQGetMessageOptions
                {
                    Options = MQC.MQGMO_WAIT,
                    WaitInterval = 5000
                };

                queue.Get(msg, getOptions);

                var data = msg.ReadString(msg.MessageLength);
                Console.WriteLine($"Received: {data}");
            }
            catch (MQException ex) when (ex.ReasonCode == MQC.MQRC_NO_MSG_AVAILABLE)
            {
                // no message available
            }

            await Task.Delay(500, stoppingToken);
        }
    }
}
